function mkid(){ return Date.now()+'-'+Math.random().toString(36).slice(2,6); }
function fmt(n){ return Number(n).toLocaleString(undefined,{minimumFractionDigits:0,maximumFractionDigits:2}); }

function catLabel(id,isIncome,isSav){

  for(const c of pool){

  }

function renderAll(){
  const mk=curMKey(), today=todayStr();
  const mExp=D.exp.filter(e=>mKey(e.date)===mk);
  const mInc=D.inc.filter(e=>mKey(e.date)===mk);
  const mSav=D.sav.filter(e=>mKey(e.date)===mk);
  const spent=mExp.reduce((s,e)=>s+Number(e.amount),0);
  const earned=mInc.reduce((s,e)=>s+Number(e.amount),0);
  const savedM=mSav.reduce((s,e)=>s+Number(e.amount),0);
  const savAll=D.sav.reduce((s,e)=>s+Number(e.amount),0);
  const bal=earned-spent-savedM;

  document.getElementById('balVal').textContent=(bal<0?'−':'')+fmt(Math.abs(bal));
  document.getElementById('balInc').textContent=fmt(earned);
  document.getElementById('balSpent').textContent=fmt(spent);
  document.getElementById('balSaved').textContent=fmt(savedM);
  document.getElementById('todayVal').textContent=fmt(D.exp.filter(e=>e.date===today).reduce((s,e)=>s+Number(e.amount),0));
  document.getElementById('mIncVal').textContent=fmt(earned);
  document.getElementById('totalSavVal').textContent=fmt(savAll);
  document.getElementById('savAllTime').textContent=fmt(savAll);

  renderList('listIncome', mInc.slice().sort((a,b)=>b.ts-a.ts), 'income');
  renderList('listToday', D.exp.filter(e=>e.date===today).sort((a,b)=>b.ts-a.ts), 'expense');
  renderList('listMonth', mExp.slice().sort((a,b)=>b.ts-a.ts), 'expense');
  renderList('listSavings', mSav.slice().sort((a,b)=>b.ts-a.ts), 'savings');

  // breakdown
  const sums={};
  mExp.forEach(e=>{ const k=e.subcat||e.cat; sums[k]=(sums[k]||0)+Number(e.amount); });
  const maxVal=Math.max(1,...Object.values(sums));
  const rows=Object.entries(sums).sort((a,b)=>b[1]-a[1]);
  document.getElementById('catBars').innerHTML=rows.length?rows.map(([id,total])=>{
    const c=catLabel(id,false,false);
    return `<div class="bar-row"><div class="bn">${c.icon} ${c.label}</div><div class="bar-track"><div class="bar-fill" style="width:${(total/maxVal*100).toFixed(0)}%"></div></div><div class="bar-amt">${fmt(total)}</div></div>`;
  }).join(''):`<div class="empty">No data yet.</div>`;

  // history
  const allKeys=new Set([...D.exp,...D.inc,...D.sav].map(e=>mKey(e.date)));
  allKeys.delete(mk);
  const sorted=[...allKeys].sort().reverse();
  document.getElementById('histList').innerHTML=sorted.length?sorted.map(k=>{
    const ex=D.exp.filter(e=>mKey(e.date)===k).reduce((s,e)=>s+Number(e.amount),0);
    const ic=D.inc.filter(e=>mKey(e.date)===k).reduce((s,e)=>s+Number(e.amount),0);
    const sv=D.sav.filter(e=>mKey(e.date)===k).reduce((s,e)=>s+Number(e.amount),0);
    const b=ic-ex-sv;
    return `<div class="hist-row"><div class="hm">${mLabel(k)}</div><div class="hn">Income ${fmt(ic)} · Spent ${fmt(ex)} · Saved ${fmt(sv)}<br>Balance <b>${b<0?'−':''}${fmt(Math.abs(b))}</b></div></div>`;
  }).join(''):`<div class="empty">No previous months yet. Data will appear here automatically at the start of each new month.</div>`;
}

}

  if(!total||!emi){ return; }
  const debt={
    id:mkid(),

function payEMI(debtId){
  const debt=D.debts.find(d=>d.id===debtId); if(!debt) return;
  D.payments.push({id:mkid(),debtId,amount:debt.emi,date:todayStr(),ts:Date.now()});
  persist('payments'); renderDebtList();
}

function delDebt(id){
  if(!confirm('Delete this loan?')) return;
  D.debts=D.debts.filter(d=>d.id!==id);
  D.payments=D.payments.filter(p=>p.debtId!==id);
  persist('debts'); persist('payments'); renderDebtList();
}

function renderDebtList(){
  const el=document.getElementById('debtList');
  if(!D.debts.length){ el.innerHTML=`<div class="empty" style="margin-bottom:12px">No loans added yet.</div>`; return; }
  el.innerHTML=D.debts.map(d=>{
    const paid=D.payments.filter(p=>p.debtId===d.id).reduce((s,p)=>s+Number(p.amount),0);
    const remaining=Math.max(0,d.total-paid);
    const pct=Math.min(100,paid/d.total*100);
    return `<div class="debt-card">
      <div class="debt-head">
        <div>
          <div class="debt-type-tag">${DEBT_LABELS[d.type]||d.type}</div>
          <div class="debt-name" style="margin-top:3px">${d.lender}</div>
        </div>
        <button class="debt-del" onclick="delDebt('${d.id}')">🗑️</button>
      </div>
      <div class="debt-bar"><div class="debt-bar-fill" style="width:${pct.toFixed(0)}%"></div></div>
      <div class="debt-nums">
        <span>Paid: ${d.currency} ${fmt(paid)}</span>
        <span>Remaining: ${d.currency} ${fmt(remaining)}</span>
      </div>
      <div class="debt-emi-row">
        <div class="debt-emi-info">
          Monthly EMI: ${d.currency} ${fmt(d.emi)}

        </div>

      </div>
    </div>`;
  }).join('');
}

function renderDocs(){
  const pid=document.getElementById('personSelect').value;
  const sec=document.getElementById('docSection');
  if(!pid){ sec.style.display='none'; return; }
  sec.style.display='';
  selDocType=null;
  document.querySelectorAll('.doc-type-btn').forEach(b=>b.classList.remove('sel'));
  document.getElementById('docForm').style.display='none';
  const myDocs=D.docs.filter(d=>d.personId===pid);
  document.getElementById('docList').innerHTML=myDocs.length?myDocs.map(d=>{
    const today=new Date(), exp=d.expiry?new Date(d.expiry):null;
    const daysLeft=exp?Math.round((exp-today)/86400000):null;
    const soonClass=daysLeft!==null&&daysLeft<90?'soon':'';
    const expText=d.expiry?(soonClass?`⚠️ Expires ${d.expiry} (${daysLeft} days)`:`Expires ${d.expiry}`):'No expiry';
    return `<div class="doc-card">
      <div class="doc-card-head">
        <div>
          <div class="doc-type-tag">${DOC_LABELS[d.type]||d.type}</div>
          <div class="doc-num">${d.number||'—'}</div>
          <div class="doc-exp ${soonClass}">${expText}</div>
        </div>
        <button class="doc-del" onclick="delDoc('${d.id}')">×</button>
      </div>
    </div>`;
  }).join(''):`<div class="empty">No documents added for this person yet.</div>`;
}

function delAppt(id){
  D.appts=D.appts.filter(a=>a.id!==id); persist('appts'); renderAppts();
}

function renderAppts(){
  const el=document.getElementById('apptList');
  const sorted=D.appts.slice().sort((a,b)=>new Date(a.datetime)-new Date(b.datetime));
  if(!sorted.length){ el.innerHTML=`<div class="empty">No appointments yet.</div>`; return; }
  el.innerHTML=sorted.map(a=>{
    const d=new Date(a.datetime);
    const dt=d.toLocaleDateString(undefined,{weekday:'short',day:'numeric',month:'short'})+' at '+d.toLocaleTimeString(undefined,{hour:'2-digit',minute:'2-digit'});
    const past=d<new Date();
    return `<div class="appt-card">
      <div class="appt-ico" style="background:${a.type==='doctor'?'#fce4ec':'#f3e5f5'}">${APPT_ICONS[a.type]||'📋'}</div>
      <div class="appt-info">
        <div class="appt-title" style="${past?'opacity:.55':''}${past?' text-decoration:line-through;':''}">${a.title}</div>
        <div class="appt-dt">${dt}${past?' (past)':''}</div>
        ${a.notes?`<div style="font-size:11px;color:var(--muted);margin-top:2px;font-weight:600">${a.notes}</div>`:''}
      </div>
      <button class="appt-del" onclick="delAppt('${a.id}')">×</button>
    </div>`;
  }).join('');
}

function exportBackup(){
  const data={app:'My Money',version:2,exportedAt:new Date().toISOString(),...D};
  const blob=new Blob([JSON.stringify(data,null,2)],{type:'application/json'});
  const url=URL.createObjectURL(blob);
  const a=document.createElement('a');
  a.href=url; a.download=`my-money-backup-${todayStr()}.json`;
  document.body.appendChild(a); a.click();
  document.body.removeChild(a); URL.revokeObjectURL(url);
}

function importBackup(input){
  const file=input.files[0]; if(!file) return;
  const reader=new FileReader();
  reader.onload=ev=>{
    try{
      const bk=JSON.parse(ev.target.result);
      const merge=confirm('Tap OK to MERGE with current data.\nTap Cancel to REPLACE all current data.');
      Object.keys(K).forEach(k=>{
        if(Array.isArray(bk[k])){
          if(merge){
            const ids=new Set(D[k].map(x=>x.id));
            D[k]=[...D[k],...bk[k].filter(x=>!ids.has(x.id))];
          } else { D[k]=bk[k]; }
          persist(k);
        }
      });
      renderAll(); alert('Backup restored!');
    }catch{ alert('Invalid backup file.'); }
    input.value='';
  };
  reader.readAsText(file);
}
